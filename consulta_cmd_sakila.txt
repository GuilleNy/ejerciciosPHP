from film, film_category , category
where film.film_id=film_category.film_id
and film_category.category_id=category.category_id

-Numero peliculas de cada categoria.

select name , count(title) as num_categoria
from film, film_category , category
where film.film_id=film_category.film_id
and film_category.category_id=category.category_id
group by name;

-Mostrar categorias con mas de 20 peliculas

select name , count(title) as num_categoria
from film, film_category , category
where film.film_id=film_category.film_id
and film_category.category_id=category.category_id
group by name
having num_categoria > 20;

- Datos de la pelicula de mayor duracion (length)

select title , rental_duration, rental_rate , release_year , length 
from film
where length=(select max(length) from film);


-Datos de los actores que han realizado mas de 10 peliculas

select first_name, last_name, count(title) 'numero'
from film , film_actor, actor
where actor.actor_id=film_actor.actor_id 
and film.film_id=film_actor.film_id
group by first_name
having count(title) > 10;
